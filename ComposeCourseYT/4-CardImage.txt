class MainActivity : ComponentActivity() {
    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContent {
            val painter = painterResource(id = R.drawable.gorsel)
            val description = "Bu bir görseldir"
            val title = "Bu görsel bir dosyadır"

            Row(modifier = Modifier.fillMaxSize().fillMaxWidth().border(2.dp, Color.Red)) {
                Box(modifier = Modifier
                    .fillMaxWidth(0.5f)
                    .padding(10.dp)
                ){
                    ImageCard(painter = painter, contentDescription = description, title = title)
                }

                Box(modifier = Modifier
                    //.fillMaxWidth(0.5f)
                    .padding(10.dp)
                ){
                    ImageCard(painter = painter, contentDescription = description, title = title)
                }
            }




        }
    }
}


@Composable //bir fonksiyonu compose yapar, compose için uygun çalıştırır
fun ImageCard(
    painter : Painter,
    contentDescription : String,
    title : String,
    modifier : Modifier = Modifier,
) {
    Card(
        modifier = Modifier.fillMaxWidth(),
        shape = RoundedCornerShape(15.dp),//yuvarlak köşe
        elevation = CardDefaults.cardElevation(
            defaultElevation = 6.dp
        ),
        //elevation = CardElevation(5.dp) // gölge verme
    ){
        Box(modifier=Modifier.height(200.dp)){
            Image(painter = painter,
                contentDescription = contentDescription,
                contentScale = ContentScale.Crop
                )
            Box(
                modifier = Modifier
                    .fillMaxSize()
                    .background(
                        Brush.verticalGradient(
                            colors = listOf(
                                Color.Transparent,
                                Color.Red
                            ),
                            startY = 300f
                        )
                    )
            )
            Box(modifier = Modifier
                .fillMaxSize()
                .padding(12.dp),
                contentAlignment = Alignment.BottomStart
            ){
                Text(title, style = TextStyle(color = Color.White, fontSize = 16.sp))
            }
        }

    }
}
